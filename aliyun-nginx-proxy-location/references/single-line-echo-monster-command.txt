} location /api/ { proxy_pass http://127.0.0.1:18790; proxy_set_header Host $host; proxy_set_header X-Real-IP $remote_addr; proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for; proxy_read_timeout 30s; } location /lab-results/ { proxy_pass http://127.0.0.1:18790; proxy_set_header Host $host; proxy_set_header X-Real-IP $remote_addr; proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for; proxy_read_timeout 30s; } location ~ ^/clinical(/.*)?$ { proxy_pass http://127.0.0.1:18790; proxy_set_header Host $host; proxy_set_header X-Real-IP $remote_addr; proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for; proxy_read_timeout 300s; add_header Cache-Control "no-cache" always; } ssl_certificate /etc/letsencrypt/live/www.hhysjt.com/fullchain.pem; ssl_certificate_key /etc/letsencrypt/live/www.hhysjt.com/privkey.pem; include /etc/letsencrypt/options-ssl-nginx.conf; ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; }' > /etc/nginx/conf.d/hhysjt.conf
